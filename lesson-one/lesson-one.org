** Dart Null Safety: the ? operator, ??, ??= and why they prevent app crashes
    #+BEGIN_SRC dart
      void main(){
        /// Nullable type — can hold null
        String? grade; // grade is null right now
        grade = 'A';   // now it has a value

        /// Null-aware operators
        String display = grade ?? 'Not graded'; // ?? = fallback if null
        print('$studentName scored $marks → $display');
        print('School: $school | Pass mark: $passMark');
      }
#+END_SRC

  void main(){

    String getGradeFeedbackIf(int score) {
      if (score >= 90 && score <= 100) {
        return 'Excellent! You got an A.';
      } else if (score >= 80 && score < 90) {
        return 'Great job! You got a B.';
      } else if (score >= 70 && score < 80) {
        return 'Good effort! You got a C.';
      } else if (score >= 0 && score < 70) {
        return 'Needs improvement. You got an F.';
      } else {
        return 'Invalid score provided.';
      }
    }

    print("Grade is: ${getGradeFeedbackIf(80)}");

    String getGradeFeedbackSwitch(int score) {                
        return switch (score) {                                                                                         
          >= 90 && <= 100 => 'Excellent! You got an A.',                                                                
          >= 80 && < 90   => 'Great job! You got a B.',              
          >= 70 && < 80   => 'Good effort! You got a C.',                                                           
          >= 0 && < 70    => 'Needs improvement. You got an F.',
          _               => 'Invalid score provided.', // Default wildcard case
        };
      }
      
      print("Grade is: ${getGradeFeedbackSwitch(90)}");  
  }

** Build a working CLI Grade Calculator by the end of class

#+BEGIN_SRC dart
  import 'dart:io';

  String calculateGrade(int marks) {
    return switch (marks) {
      >= 80 && <= 100 => 'A+',
      >= 70 && < 80   => 'A',
      >= 60 && < 70   => 'B',
      >= 50 && < 60   => 'C',
      >= 40 && < 50   => 'D',
      >= 0 && < 40    => 'F',
      _               => 'Invalid Score', // Handles negative numbers or > 100
    };
  }

  void main() {
    final students = <Map<String, dynamic>>[];

    print('=== Student Grade Calculator ===');
    for (int i = 1; i <= 3; i++) {
      stdout.write('Enter name for student $i: ');
      final name = stdin.readLineSync() ?? 'Unknown';
      
      stdout.write('Enter marks (0-100): ');
      // int.tryParse prevents crashes if the user types letters
      final marks = int.tryParse(stdin.readLineSync() ?? '0') ?? 0; 
      
      students.add({
        'name': name, 
        'marks': marks,
        'grade': calculateGrade(marks)
      });
    }

    print('\n--- Report Card ---');
    for (var s in students) {
      final status = s['marks'] >= 40 ? '✓ PASS' : '✗ FAIL';
      print('${s["name"]}: ${s["marks"]} → ${s["grade"]}  $status');
    }
    
    final avg = students.map((s) => s['marks'] as int).reduce((a, b) => a + b) / students.length;
    print('\nClass Average: ${avg.toStringAsFixed(1)}');
  }
#+END_SRC
